01401   SUBROUTINE BJL(L,X,JL)

01408     !!== zqhuang@astro.utoronto.ca                                                 ==!!

01411     REAL*8 X,JL
01412     REAL*8 AX,AX2
01413     REAL,PARAMETER::LN2=0.6931471805599453094D0
01414     REAL,PARAMETER::ONEMLN2=0.30685281944005469058277D0
