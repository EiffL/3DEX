02034           dalm(1:2, m:nlmax ) = 0.0_dp
02035 
02036           do ithl = 0, uchk - lchk
02037              l_min = l_min_ylm(m, sth(ithl))
02038              if (keep_it(ithl) .and. nlmax >= l_min) then ! avoid un-necessary calculations (EH, 09-2001)
02039                 ! determine lam_mm
02040                 call compute_lam_mm(mfac(m), m, sth(ithl), lam_mm, corfac, scalem)
02041 
02042                 !           ---------- l = m ----------
02043                 par_lm = 1
02044 
02045                 php = phasl_n(m,m,ithl) + phasl_s(m,m,ithl) ! sum  (if (l+m) even)
02046                 phm = phasl_n(m,m,ithl) - phasl_s(m,m,ithl) ! diff (if (l+m) odd)
02047                 phas_sd(-1:0) =  (/ real(phm, kind=dp), aimag(phm) /)
02048                 phas_sd(1:2) =  (/ real(php, kind=dp), aimag(php) /)
02049                 
02050                if (m >= l_min) then
02051                   lam_lm = lam_mm * corfac !Actual lam_mm 
02052                   dalm(1:2, m) = dalm(1:2, m) + phas_sd(par_lm:par_lm+1) *lam_lm
02053                endif
02054

02056                 !           ---------- l > m ----------
02057                 lam_0 = 0.0_dp
02058                 lam_1 = 1.0_dp
02059                 scalel=0
02060                 cth_ring = cth(ithl)
02061                 lam_2 = cth_ring * lam_1 * recfac(0,m)
02062                 
02063                 do l = m+1, nlmax
02064                 
02065                    php = phasl_n(l,m,ithl) + phasl_s(l,m,ithl) ! sum  (if (l+m) even)
02066                    phm = phasl_n(l,m,ithl) - phasl_s(l,m,ithl) ! diff (if (l+m) odd)
02067                    phas_sd(-1:0) =  (/ real(phm, kind=dp), aimag(phm) /)
02068                    phas_sd(1:2) =  (/ real(php, kind=dp), aimag(php) /)
02069                    
02070                    par_lm = - par_lm  ! = (-1)^(l+m)
02071                    
02072                    if (l >= l_min) then
02073                       lam_lm = lam_2 * corfac * lam_mm
02074                       dalm(1:2, l) = dalm(1:2, l) &
02075                            &       +  phas_sd(par_lm:par_lm+1) *lam_lm
02076                    endif
02077 
02078                    lam_0 = lam_1 * recfac(1,l-1)
02079                    lam_1 = lam_2
02080                    lam_2 = (cth_ring * lam_1 - lam_0) * recfac(0,l)
02081                    if (abs(lam_2) > OVFLOW) then
02082                       lam_1 = lam_1*UNFLOW
02083                       lam_2 = lam_2*UNFLOW
02084                       scalel= scalel + 1
02085                       corfac= rescale_tab(max(scalem+scalel,RSMIN))
02086                    elseif (abs(lam_2) < UNFLOW) then
02087                       lam_1 = lam_1*OVFLOW
02088                       lam_2 = lam_2*OVFLOW
02089                       scalel= scalel - 1
02090                       corfac= rescale_tab(max(scalem+scalel,RSMIN))
02091                    endif
02092 
02093                 enddo ! loop on l
02094              endif ! test on cut sky and nlmax
02095           enddo ! loop on ithl
02096           do l = m, nlmax
02097              almn(1, l, m) = almn(1, l, m) + cmplx(dalm(1, l), dalm(2, l), kind=DP)
02098           enddo
02099        enddo ! loop on m
02100 !$OMP end do
02101        if (do_openmp()) then
02102           deallocate (recfac,dalm)
02103        endif
02104 !$OMP end parallel
02105     enddo ! loop on chunks
02106 
02107     !     --------------------
02108     !     free memory and exit
02109     !     --------------------
02110 
02111     deallocate(mfac)
02112     if (.not.do_openmp()) then
02113        deallocate (ring,recfac,dalm)
02114         deallocate(phas_n,phas_s)
02115     endif
02116     RETURN

02176     SUBROUTINE almn2rmap(map, almn, rho, nsmax, nnmax, nlmax, nmmax, kln, cln, plm)
02177 
02178  IMPLICIT NONE
02179     
02180  real(DP)             :: rho
02181  integer(I4B), intent(IN)                     :: nsmax, nnmax, nlmax, nmmax
02182  complex(DPC), intent(IN),  dimension(1:nnmax,0:nlmax,0:nmmax)      :: almn
02183  real(DP),     intent(OUT), dimension(0:(12_i8b*nsmax)*nsmax-1,1:1) :: map
02184  real(DP),     intent(IN),  dimension(0:( nsmax*(nmmax+1)*(2*nlmax-nmmax+2) ) )   :: plm
02185     
02186  integer(I4B) :: l, m, ith, n
02187  integer(I8B) :: istart_south, istart_north, npix
02188  integer(I4B) :: nrings, nphmx
02189     
02190  integer(I8B) :: n_lm, n_plm, i_mm
02191  real(DP),     dimension(0:nlmax,1:nnmax)  :: cln, kln,jlns
02192  complex(DPC), dimension(-1:1)             :: b_ns
02193  real(DP),     dimension(:,:), allocatable :: dalm
02194  integer(i4b)                              :: ll, l_min, l_start
02195  real(DP)                                  :: cth
02196  complex(DPC), dimension(:,:), allocatable :: b_north, b_south
02197  real(DP),     dimension(:),   allocatable :: ring
02198  integer(i4b)             :: nchunks, chunksize, ichunk, lchk, uchk, ithl, nphl
02199  integer(i4b), parameter :: SMAXCHK = 50
02200  integer(I8B), dimension(0:SMAXCHK-1) :: startpix
02201  integer(I4B), dimension(0:SMAXCHK-1) :: nph, kphi0
02202  real(DP),     dimension(0:SMAXCHK-1) :: sth
02203      
02204  character(LEN=*), parameter :: code = 'ALM2MAP'
02205  integer(I4B) :: status
02206  !=======================================================================
02207     
02208  ! Healpix definitions
02209  nrings = 2*nsmax           ! number of isolatitude rings on N. hemisphere + equat
02210  npix   = (12_I8B*nsmax)*nsmax    ! total number of pixels on the sphere
02211  nphmx  = 4*nsmax           ! maximum number of pixels/ring
02212  n_lm   = ((nmmax+1_I8B)*(2*nlmax-nmmax+2))/2 !number of (l,m) with m in[0,M] and l in [m,L]
02213  n_plm  = n_lm * nrings
02214     
02215  !     --- allocates space for arrays ---
02216  nchunks   = nrings/SMAXCHK + 1  ! number of chunks
02217  chunksize = (nrings+nchunks-1)/nchunks ! <= SMAXCHK
02218     
02219  allocate(b_north(0:nmmax, 0:chunksize-1),stat = status)
02220  call assert_alloc(status,code,'b_north')
02221     
02222  allocate(b_south(0:nmmax, 0:chunksize-1),stat = status)
02223  call assert_alloc(status,code,'b_south')
02224     
02225  if (.not. do_openmp()) then
02226     allocate(dalm(0:1,0:nlmax), stat = status)
02227     call assert_alloc(status,code,'dalm')
02228     allocate(ring(0:nphmx-1),stat = status)
02229     call assert_alloc(status,code,'ring')
02230  endif
02231  
02232  
02233  !     ------------ Compute required bessel functions ----------------
02234         
02235  CALL gen_jln(jlns, kln, rho, nnmax, nlmax)
02236  
02237  
02238  !     ------------ initiate variables and arrays ----------------
02239     
02240  map = 0.0 ! set the whole map to zero
02241     
02242  ! loop on chunks
02243  do ichunk = 0, nchunks-1
02244     lchk = ichunk * chunksize + 1
02245     uchk = min(lchk+chunksize - 1, nrings)
02246     
02247     do ith = lchk, uchk
02248        ithl = ith - lchk !local index
02249        ! get pixel location information
02250        call get_pixel_layout(nsmax, ith, cth, sth(ithl), nph(ithl), startpix(ithl), kphi0(ithl))
02251     enddo
02252     !        -----------------------------------------------------
02253     !        for each theta, and each m, computes
02254     !        b(m,theta) = sum_over_l>m (lambda_l_m(theta) * a_l_m)
02255     !        ------------------------------------------------------
02256     
02257     b_north(:,:) = 0_dpc ! pad with zeros
02258     b_south(:,:) = 0_dpc
02259     
02260     !$OMP parallel default(none) NUM_THREADS(8) &
02261     !$OMP shared(nlmax, nmmax, sth, lchk, uchk, plm, b_north, b_south, n_lm, jlns, cln, almn, nnmax) &
02262     !$OMP private(dalm, b_ns, status, m, ith, ithl, l_min, l_start, l, ll, i_mm)
02263     
02264     if (do_openmp()) then
02265        allocate(dalm(0:1,0:nlmax), stat = status)
02266        call assert_alloc(status,code,'dalm')
02267     endif
02268     
02269     
02270     !$OMP do schedule(dynamic,1)
02271     do m = 0, nmmax
02272        dalm = 0.0_dp
02273        ! extract needed alm under memory and CPU efficient form
02274        do ll = m, nlmax
02275         do n = 0, nnmax
02276     dalm(0,ll) = dalm(0,ll) + real(almn(n,ll,m),kind=dp) * jlns(ll,n) * cln(ll,n)
02277      dalm(1,ll) = dalm(1,ll) + aimag(almn(n,ll,m)) * jlns(ll,n) * cln(ll,n)
02278     enddo
02279        enddo
02280        do ithl = 0, uchk - lchk
02281    l_min = l_min_ylm(m, sth(ithl))
02282    if (nlmax >= l_min) then ! skip calculations when Ylm too small
02283       ith = ithl + lchk
02284       i_mm = n_lm * (ith-1) + ((2_I8B*nlmax + 3 - m)*m)/2 ! location of Ym,m for ring ith
02285     
02286       !           ---------- l = m ----------
02287     
02288       if (m >= l_min) then
02289          b_ns( 1) = cmplx(plm(i_mm) * dalm(0,m), plm(i_mm) * dalm(1,m), kind=DP)
02290          b_ns(-1) = 0.0_dpc
02291       else
02292          b_ns = 0.0_dpc
02293       endif
02294     
02295       !           ---------- l > m ----------
02296       l_start = max(m+1, l_min) ! odd values of (l+m)
02297       if (mod(m+l_start,2) == 0) l_start = l_start+1
02298       do l = l_start, nlmax, 2
02299          b_ns(-1) = b_ns(-1) + &
02300        & cmplx(plm(i_mm+l-m) * dalm(0,l), plm(i_mm+l-m)*dalm(1,l), kind=DP)
02301       enddo ! loop on l
02302     
02303       l_start = max(m+2, l_min) ! even values of (l+m)
02304       if (mod(m+l_start,2) == 1) l_start = l_start+1
02305       do l = l_start, nlmax, 2
02306          b_ns(1)  = b_ns(1) + &
02307        & cmplx(plm(i_mm+l-m) * dalm(0,l), plm(i_mm+l-m)*dalm(1,l),kind=DP) 
02308       enddo ! loop on l
02309     
02310       b_north(m,ithl) = b_ns(1) + b_ns(-1)
02311       b_south(m,ithl) = b_ns(1) - b_ns(-1)
02312    endif ! test on nlmax
02313        enddo ! loop on rings (ithl)
02314     enddo ! loop on m
02315     !$OMP end do
02316     
02317     if (do_openmp()) then
02318        deallocate (dalm)
02319     endif
02320     !$OMP end parallel
02321     
02322     !$OMP parallel default(none) NUM_THREADS(8) &
02323     !$OMP  shared(nsmax, nlmax, nmmax, npix, nrings, nphmx, &
02324     !$OMP      lchk, uchk, b_north, b_south, nph, startpix, kphi0, map) &
02325     !$OMP  private(ithl, nphl, istart_north, istart_south, &
02326     !$OMP      ith, ring, status)
02327     if (do_openmp()) then
02328        allocate(ring(0:nphmx-1),stat = status)
02329        call assert_alloc(status,code,'ring')
02330     endif
02331     !$OMP do schedule(dynamic,1)
02332     do ithl = 0, uchk - lchk
02333        nphl = nph(ithl)
02334        istart_north = startpix(ithl)
02335        istart_south = npix-istart_north-nphl
02336        ith  = ithl + lchk
02337        !        ---------------------------------------------------------------
02338        !        sum_m  b(m,theta)*exp(i*m*phi)   -> f(phi,theta) by FFT
02339        !        ---------------------------------------------------------------
02340        call ring_synthesis(nsmax,nlmax,nmmax,b_north(0,ithl),nphl,ring,kphi0(ithl))   ! north hemisph. + equator
02341        map(istart_north:istart_north+nphl-1,1) = map(istart_north:istart_north+nphl-1,1) + ring(0:nphl-1)
02342     
02343        if (ith < nrings) then
02344    call ring_synthesis(nsmax,nlmax,nmmax,b_south(0,ithl),nphl,ring,kphi0(ithl)) ! south hemisph. w/o equat
02345    map(istart_south:istart_south+nphl-1,1) = map(istart_south:istart_south+nphl-1,1) + ring(0:nphl-1)
02346        endif
02347     enddo ! loop on ithl
02348     !$OMP end do
02349     if (do_openmp()) then
02350        deallocate(ring)
02351     endif
02352     !$OMP end parallel
02353  enddo    ! loop on chunks
02354     
02355  !     --------------------
02356  !     free memory and exit
02357  !     --------------------
02358  if (.not.do_openmp()) then
02359     deallocate (ring,dalm)
02360  endif
02361  deallocate(b_north, b_south)
02362  return
